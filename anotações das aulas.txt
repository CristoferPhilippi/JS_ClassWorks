// + função literal

function somar(a, b) {
  console.log(a + b);
}
somar(1, 2);

// + Função anônima (sem nome)

const somar = function (a, b) {
  console.log(a + b);
};
somar(1, 2);

//+ Função arrow (seta)

const somar = (a, b) => {
  console.log(a + b);
};
somar(1, 2);

//+Função arrow (retorno implícito)

const somar = (a, b) => console.log(a + b);
somar(1, 2);

// ou
const somar = (a, b) => a + b;
let resultado = somar(1, 2);
console.log(resultado);

function verificarDisponibilidade(quartos, ocupados) {
  let res = quartos - ocupados;
  console.log("Disponíveis: " + res);
}

let quartos = 20;
let ocupados = 15;
verificarDisponibilidade(quartos, ocupados);

//orientado a objeto

const hotel = {
  quartos: 50,
  ocupados: 30,
  verificarDisponibilidade: function () {
    let res = this.quartos - this.ocupados;
    console.log("Disponíveis: " + res);
  },
};

hotel.ocupados = 5;
hotel.verificarDisponibilidade();

//Criando classes (mais simples) template

class Hotel {
  constructor() {
    this.quartos = 20;
    this.ocupados = 10;
  }

  verificarDisponibilidade() {
    let res = this.quartos - this.ocupados;
    return "Disponíveis: " + res;
  }
}

const hotel = new Hotel();
hotel.verificarDisponibilidade();
